2.4 계산 로직

정의: 오늘 resolved된 failed/blocked 건들의 평균 (발견→해결) 시간

// Avg Response Time 계산 (resolved_at 기반)
const resolvedToday = results.filter(r =>
  r.resolved_at &&
  r.resolved_at >= todayISO &&
  r.status !== 'failed' &&
  r.status !== 'blocked'  // 현재 상태가 passed/retest (= 해결됨)
);

if (resolvedToday.length === 0) return '—';

const totalMs = resolvedToday.reduce((sum, r) => {
  return sum + (new Date(r.resolved_at).getTime() - new Date(r.created_at).getTime());
}, 0);

const avgMs = totalMs / resolvedToday.length;
const avgMins = avgMs / 60000;

// 포맷팅
if (avgMins < 60) return `${Math.round(avgMins)}m`;
if (avgMins < 1440) return `${(avgMins / 60).toFixed(1)}h`;
return `${(avgMins / 1440).toFixed(1)}d`;

데이터현재변경

성능 권장: 52주분 raw 데이터를 클라이언트로 가져오면 수천 건이 될 수 있음. Supabase RPC 함수로 일별 count를 서버사이드에서 집계하는 것을 권장:

-- Supabase RPC 함수 (권장)
CREATE OR REPLACE FUNCTION get_daily_activity_counts(
  p_project_ids UUID[],
  p_start_date DATE
) RETURNS TABLE (activity_date DATE, activity_count BIGINT) AS $$
  SELECT d.activity_date, COALESCE(r.cnt, 0) + COALESCE(tc.cnt, 0)
  FROM generate_series(p_start_date, CURRENT_DATE, '1 day') d(activity_date)
  LEFT JOIN (
    SELECT DATE(created_at) AS dt, COUNT(*) AS cnt
    FROM test_results tr
    JOIN test_runs trun ON tr.run_id = trun.id
    WHERE trun.project_id = ANY(p_project_ids)
      AND tr.created_at >= p_start_date
    GROUP BY DATE(created_at)
  ) r ON r.dt = d.activity_date
  LEFT JOIN (
    SELECT DATE(created_at) AS dt, COUNT(*) AS cnt
    FROM test_cases
    WHERE project_id = ANY(p_project_ids)
      AND created_at >= p_start_date
    GROUP BY DATE(created_at)
  ) tc ON tc.dt = d.activity_date
$$ LANGUAGE SQL STABLE;

호출: supabase.rpc('get_daily_activity_counts', { p_project_ids: projectIds, p_start_date: oneYearAgo })
