13. Database Schema

-- Add lifecycle_status to test_cases table
ALTER TABLE test_cases
  ADD COLUMN lifecycle_status text
  NOT NULL DEFAULT 'draft'
  CHECK (lifecycle_status IN ('draft', 'active', 'deprecated'));

-- Index for filter queries
CREATE INDEX idx_tc_lifecycle ON test_cases(project_id, lifecycle_status);

-- History tracking (existing table)
-- test_case_history already supports:
--   action_type: 'status_changed'
--   field_name: 'lifecycle_status'
--   old_value / new_value: 'draft'|'active'|'deprecated'

14. API / Supabase Queries

// Update lifecycle status (single TC)
async function setLifecycleStatus(tcId: string, newStatus: string) {
  // 1. Optimistic UI update
  setTestCase(prev => ({ ...prev, lifecycle_status: newStatus }));

  // 2. DB update
  const { error } = await supabase
    .from('test_cases')
    .update({ lifecycle_status: newStatus })
    .eq('id', tcId);

  if (error) {
    // 3. Rollback on failure
    setTestCase(prev => ({ ...prev, lifecycle_status: oldStatus }));
    toast.error('상태 변경에 실패했습니다.');
    return;
  }

  // 4. Record history
  await supabase.from('test_case_history').insert({
    test_case_id: tcId,
    user_id: user.id,
    action_type: 'status_changed',
    field_name: 'lifecycle_status',
    old_value: oldStatus,
    new_value: newStatus,
  });
}

// Bulk update
async function bulkSetLifecycleStatus(tcIds: string[], newStatus: string) {
  const { error } = await supabase
    .from('test_cases')
    .update({ lifecycle_status: newStatus })
    .in('id', tcIds);
  // + bulk history insert for each TC
}

// Run TC selection query
const { data } = await supabase
  .from('test_cases')
  .select('*, lifecycle_status')
  .eq('project_id', projectId)
  .neq('lifecycle_status', 'deprecated');  // Always exclude deprecated from Run selection

// Check deprecated TCs in existing Run
const deprecatedInRun = await supabase
  .from('run_test_cases')
  .select('test_case_id, test_cases!inner(lifecycle_status)')
  .eq('run_id', runId)
  .eq('test_cases.lifecycle_status', 'deprecated');

15. Accessibility & i18n

Accessibility

  • All badges are <button> elements with aria-haspopup="listbox" when clickable
  • Dropdown panel uses role="listbox", items use role="option" with aria-selected
  • Context menu uses role="menu" / role="menuitem"
  • Toggle switch uses role="switch" with aria-checked
  • Focus trap on confirm dialog, initial focus on Cancel button (safety)
  • Color is never sole indicator: badges always include icon + text label
  • Deprecated row opacity (0.55/0.65) still meets WCAG AA contrast with badge text
