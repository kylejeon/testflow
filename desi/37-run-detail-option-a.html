§5 — React Component Architecture
// Shared base component with context-specific rendering

interface DetailPanelProps {
  context: 'testcase' | 'run';  // determines which features show
  testCase: TestCase;
  runExecution?: RunExecution;  // only in run context
}

// Steps area — renders differently based on context
function StepsArea({ context, steps, stepResults }) {
  return (
    <div className="dp-steps-area">
      {steps.map(step => (
        <StepRow key={step.id} step={step}
          result={stepResults?.[step.id]}
          /* ★ Mini buttons only in run context */
          showResultButtons={context === 'run'}
        />
      ))}
    </div>
  );
}

// StepRow with conditional mini buttons
function StepRow({ step, result, showResultButtons }) {
  const stateClass = result === 'passed' ? 'step-passed'
                    : result === 'failed' ? 'step-failed'
                    : '';
  return (
    <div className={`dp-step ${stateClass}`}>
      <StepNum />
      <StepContent />
      {showResultButtons && (
        <div className="dp-step-actions">
          <MiniPassBtn />
          <MiniFailBtn />
        </div>
      )}
    </div>
  );
}
Key architectural decision: Use a single DetailPanel component with a context prop rather than two separate components. This ensures consistent styling while allowing run-specific features (step results, Quick Actions, Pass & Next) to render conditionally.
§6 — Step Results Data Model
// New table: step_results
CREATE TABLE step_results (
  id          uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  execution_id uuid REFERENCES executions(id),
  step_id     uuid REFERENCES test_steps(id),
  status      text CHECK (status IN ('passed','failed','blocked')),
  note        text,
  created_at  timestamptz DEFAULT now(),
  created_by  uuid REFERENCES profiles(id)
);

// Index for fast lookup per execution
CREATE INDEX idx_step_results_execution
  ON step_results(execution_id, step_id);
