1 아키텍처 개요 & Panel Layout

Option A의 핵심: Meta grid + Steps + Attachments를 탭 위에 항상 노출하고, 기존 "Details" 탭을 제거합니다. TC Detail과 Run Detail은 동일한 레이아웃 골격을 공유하되, Run context에서만 Quick Actions bar, Step-level result 버튼, Step progress가 추가됩니다.

2 공통 컴포넌트 설계 — DetailPanel

하나의 DetailPanel 컴포넌트를 만들어 context prop으로 TC/Run 분기합니다. 이렇게 하면 스타일·레이아웃 일관성을 유지하면서 Run 전용 기능(Step results, Quick Actions)을 조건부로 렌더링할 수 있습니다.

Interface Definition
// src/components/DetailPanel/types.ts

export interface DetailPanelProps {
  /** 'testcase' = TC 목록에서, 'run' = Run Detail에서 */
  context: 'testcase' | 'run';

  /** 공통: 테스트 케이스 데이터 */
  testCase: {
    id: string;
    title: string;
    description?: string;
    customId?: string;        // TC-001 형식
    folder?: string;
    priority: string;
    tags?: string;            // comma-separated
    assignee?: string;
    createdAt: string;
    steps?: string;            // JSON 또는 newline-separated
    expected_result?: string;
    precondition?: string;     // ★ Phase 0 Bug fix: 반드시 전달
    attachments?: { name: string; url: string; size: number }[];
  };

  /** Run context only */
  runExecution?: {
    runStatus: TestStatus;
    stepResults?: Record<string, 'passed' | 'failed' | 'blocked'>;
    lastRunName?: string;
  };

  /** Callbacks */
  onClose: () => void;
  onEdit?: () => void;              // TC context
  onDelete?: () => void;            // TC context
  onStatusChange?: (status: TestStatus) => void;   // Run context
  onPassAndNext?: () => void;       // Run context
  onAddResult?: () => void;        // Run context
  onPrev?: () => void;
  onNext?: () => void;
  onStepResult?: (stepId: string, status: 'passed'|'failed'|null) => void;  // Run context
}
Component Structure (JSX)
// src/components/DetailPanel/DetailPanel.tsx

export function DetailPanel({ context, testCase, runExecution, ...props }: DetailPanelProps) {
  const [activeTab, setActiveTab] = useState<'comments'|'results'|'issues'|'history'>('comments');
  const [stepsCollapsed, setStepsCollapsed] = useState(false);

  const steps = parseSteps(testCase.steps);
  const isRun = context === 'run';

  return (
    <div className="w-[500px] min-w-[500px] bg-white border-l border-gray-200 flex flex-col overflow-hidden">

      {/* ① Header */}
      <DPHeader testCase={testCase} onClose={props.onClose} />

      {/* ② Quick Actions — Run Only */}
      {isRun && runExecution && (
        <DPQuickActions
          status={runExecution.runStatus}
          onStatusChange={props.onStatusChange}
          onAddResult={props.onAddResult}
          onPassAndNext={props.onPassAndNext}
          onPrev={props.onPrev}
          onNext={props.onNext}
        />
      )}

      {/* ③ Meta Grid — Always visible */}
      <DPMeta testCase={testCase} />

      {/* ④ Steps Toggle */}
      <DPStepsToggle
        collapsed={stepsCollapsed}
        onClick={() => setStepsCollapsed(!stepsCollapsed)}
        stepCount={steps.length}
        attachmentCount={testCase.attachments?.length || 0}
        {/* Run: show progress instead of counts */}
        passedStepCount={isRun ? countPassedSteps(runExecution?.stepResults) : undefined}
        context={context}
      />

      {/* ⑤ Steps Area — Collapsible, max 40vh */}
      {!stepsCollapsed && (
        <div className="max-h-[40vh] overflow-y-auto px-5 py-3.5 border-b border-gray-200 flex-shrink-0">
          {testCase.precondition && <Precondition text={testCase.precondition} />}
          <StepsList
            steps={steps}
            expectedResults={testCase.expected_result}
            stepResults={isRun ? runExecution?.stepResults : undefined}
            showResultButtons={isRun}
            onStepResult={props.onStepResult}
          />
          {testCase.attachments && <Attachments files={testCase.attachments} />}
        </div>
      )}

      {/* ⑥ Tabs */}
      <DPTabBar tabs={['comments','results','issues','history']} active={activeTab} onChange={setActiveTab} />

      {/* ⑦ Tab Body — fills remaining space */}
      <div className="flex-1 overflow-y-auto px-5 py-3.5">
        {activeTab === 'comments' && <CommentsTab />}
        {activeTab === 'results'  && <ResultsTab />}
        {activeTab === 'issues'   && <IssuesTab />}
        {activeTab === 'history'  && <HistoryTab />}
      </div>

      {/* ⑧ Footer — TC Only */}
      {!isRun && (
        <div className="px-5 py-3 border-t border-gray-200 flex gap-2 flex-shrink-0">
          <button className="flex-1 ... bg-indigo-500 text-white" onClick={props.onEdit}>Edit</button>
          <button className="flex-1 ... text-red-500 border-red-300" onClick={props.onDelete}>Delete</button>
        </div>
      )}
    </div>
  );
}

// StepRow.tsx — Run context

function StepRow({ step, result, showResultButtons, onStepResult }: StepRowProps) {
  const stateClass = result === 'passed' ? 'bg-green-50 border border-green-200'
                    : result === 'failed' ? 'bg-red-50 border border-red-200'
                    : 'bg-gray-50';

  const numClass = result === 'passed' ? 'bg-green-500 text-white'
                  : result === 'failed' ? 'bg-red-500 text-white'
                  : 'bg-indigo-100 text-indigo-700';

  const handleClick = (status: 'passed'|'failed') => {
    // Toggle: 이미 활성 상태면 null(untested)로 되돌림
    onStepResult?.(step.id, result === status ? null : status);
  };

  return (
    <div className={`flex gap-2 rounded-lg p-2 items-start ${stateClass}`}>
      <div className={`w-5 h-5 rounded-full flex items-center justify-center text-[0.625rem] font-bold ${numClass}`}>
        {step.order}
      </div>
      <div className="flex-1 min-w-0">
        <p className="text-xs text-gray-700 leading-relaxed">{step.text}</p>
        {step.expected && (
          <p className="text-[0.6875rem] text-green-600 mt-1 flex items-start gap-1">
            <i className="ri-checkbox-circle-line text-xs flex-shrink-0 mt-0.5" />
            {step.expected}
          </p>
        )}
      </div>
      {showResultButtons && (
        <div className="flex gap-[3px] flex-shrink-0 mt-0.5">
          <MiniBtn type="pass" active={result==='passed'} onClick={() => handleClick('passed')} />
          <MiniBtn type="fail" active={result==='failed'} onClick={() => handleClick('failed')} />
        </div>
      )}
    </div>
  );
}

9 새 DB 테이블: step_results

Step-level result 기능을 위해 새 Supabase 테이블이 필요합니다.

Schema
-- Step-level results for Run context
CREATE TABLE step_results (
  id            uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  execution_id  uuid REFERENCES executions(id) ON DELETE CASCADE,
  step_index    integer NOT NULL,  -- 0-based step order
  status        text CHECK (status IN ('passed', 'failed', 'blocked')),
  note          text,
  created_at    timestamptz DEFAULT now(),
  created_by    uuid REFERENCES profiles(id)
);

-- Index for fast lookup
CREATE UNIQUE INDEX idx_step_results_exec_step
  ON step_results(execution_id, step_index);

-- RLS: Same policy as executions table
ALTER TABLE step_results ENABLE ROW LEVEL SECURITY;

-- Enable Supabase Realtime for live updates
ALTER PUBLICATION supabase_realtime ADD TABLE step_results;
Note: execution_id는 현재 run_results 또는 test_results 테이블의 ID를 참조합니다. 실제 테이블명은 현재 스키마에 맞게 조정하세요. step_index는 steps 배열의 인덱스(0-based)입니다.
