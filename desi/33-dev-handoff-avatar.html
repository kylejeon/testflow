§2 Avatar Component — Props & API

// src/components/ui/Avatar.tsx import { getAvatarColor, getInitials } from './avatar-utils'; interface AvatarUser { id: string; full_name?: string | null; email?: string | null; avatar_url?: string | null; } interface AvatarProps { user: AvatarUser; size: 'xs' | 'sm' | 'md' | 'lg' | 'xl'; status?: 'online' | 'offline' | 'busy' | 'away'; showUpload?: boolean; // Only for 'xl' — shows camera overlay on hover className?: string; onClick?: () => void; } export function Avatar({ user, size, status, showUpload, className, onClick }: AvatarProps) { const initials = getInitials(user.full_name || user.email || ''); const bgColor = getAvatarColor(user.id); const [imgError, setImgError] = useState(false); const hasPhoto = user.avatar_url && !imgError; return ( <div className={cn('av', `av-${size}`, showUpload && 'av-upload', className)} style={{ background: hasPhoto ? 'transparent' : bgColor }} role="img" aria-label={`${user.full_name || 'User'} avatar`} onClick={onClick} > {hasPhoto ? <img src={user.avatar_url!} alt="" onError={() => setImgError(true)} /> : initials || '?'} {status && <span className={`av-status ${status}`} aria-label={status} />} {showUpload && ( <div className="av-overlay"><i className="ri-camera-line" /></div> )} </div> ); } // ── AvatarStack sub-component ── interface AvatarStackProps { users: AvatarUser[]; size: 'xs' | 'sm' | 'md'; max?: number; // default 4 } export function AvatarStack({ users, size, max = 4 }: AvatarStackProps) { const visible = users.slice(0, max); const overflow = users.length - max; return ( <div className="av-stack"> {visible.map(u => <Avatar key={u.id} user={u} size={size} />)} {overflow > 0 && <span className={`av-overflow av-${size}`}>+{overflow}</span>} </div> ); }
Import 패턴: 모든 페이지에서 import { Avatar, AvatarStack } from '@/components/ui/Avatar';로 사용합니다. 기존 인라인 아바타 마크업을 모두 이 컴포넌트 호출로 교체합니다.

§4 Color System — 10색 팔레트 + 해시 알고리즘

// src/components/ui/avatar-utils.ts export const AVATAR_COLORS = [ '#6366F1', // 0 Indigo (brand primary) '#8B5CF6', // 1 Violet (brand accent) '#EC4899', // 2 Pink '#10B981', // 3 Emerald '#F59E0B', // 4 Amber '#06B6D4', // 5 Cyan '#EF4444', // 6 Red '#14B8A6', // 7 Teal '#F97316', // 8 Orange '#3B82F6', // 9 Blue ] as const; /** * Returns a deterministic color for a userId. * Same userId → same color across all screens. * Uses djb2 hash for even distribution. */ export function getAvatarColor(userId: string): string { let hash = 0; for (let i = 0; i < userId.length; i++) { hash = ((hash << 5) - hash + userId.charCodeAt(i)) | 0; } return AVATAR_COLORS[Math.abs(hash) % AVATAR_COLORS.length]; } /** * Extracts 1-2 character initials from a name or email. * "Kim Yong Hyuk" → "KH" (first + last) * "alice@test.com" → "A" * "" → "?" */ export function getInitials(input: string): string { if (!input) return '?'; const name = input.includes('@') ? input.split('@')[0] : input; const parts = name.trim().split(/\s+/); if (parts.length === 1) return parts[0][0].toUpperCase(); return (parts[0][0] + parts[parts.length - 1][0]).toUpperCase(); }
Critical: 기존 코드에서 인라인 style="background:#6366F1" 등으로 하드코딩된 색상을 모두 제거하고 getAvatarColor(user.id)로 교체해야 합니다. 현재 같은 사용자가 페이지마다 다른 색상인 문제의 근본 원인입니다.

§9 Edge Cases

케이스처리
이름 없음 (null/empty)"?" 텍스트 + bg #CBD5E1
Email만 있는 사용자alice@test.com → "A" (@ 앞 첫 글자)
단일 이름"Alice" → "A"
3+ 단어 이름"Kim Yong Hyuk" → "KH" (첫+끝)
비라틴 문자"김영혁" → "김영" (그대로 2자)
이미지 로드 실패onError → imgError=true → 이니셜 fallback
Deleted/deactivated usergrey silhouette SVG + bg #E2E8F0
userId 없음 (비정상)getAvatarColor('') → index 0 (#6366F1)

§10 QA Checklist

  • 동일 사용자 색상 일관성: 같은 유저가 Header / Table / Team Card / Comment에서 모두 동일 색상인지 확인
  • 크기 통일: Header 아바타가 정확히 32px인지 (기존 28px 페이지가 없는지)
  • Table assignee: 모든 테이블 셀에서 24px 고정인지 (1.5rem~1.75rem 혼용 없는지)
  • Feed avatar: Activity feed에서 24px로 변경 후 레이아웃 깨지지 않는지
  • Photo fallback: avatar_url을 잘못된 URL로 세팅 후 이니셜로 정상 전환되는지
  • Stack overflow: 5명 이상 팀에서 "+N" 배지 정상 표시되는지
  • Status dot: Team Activity에서 online/offline 정상 표시, 테이블에서는 미표시인지
  • Empty name: full_name이 null인 유저 초대 시 "?" 표시되는지
  • 비라틴 문자: 한국어/일본어 이름에서 이니셜 정상 추출되는지
  • Upload: Profile Settings에서 2MB 초과 파일 업로드 시 에러 메시지 표시
  • Accessibility: 스크린 리더로 각 아바타의 aria-label 정상 읽히는지
  • CSS 정리: 기존 6개 클래스 삭제 후 어디에서도 스타일 깨짐 없는지
