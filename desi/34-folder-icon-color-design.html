§3 자동 컬러 배정 로직
기본값 개선: 기존에는 모든 새 폴더가 gray로 생성되어 사이드바가 회색 일색이었습니다.

새 로직: 새 폴더 생성 시, 기존 폴더들이 사용하지 않는 색상 중 순서대로 배정합니다.
모든 색상이 사용 중이면 가장 적게 사용된 색상을 선택합니다.
사용자가 수동으로 색상을 변경하면 해당 선택이 우선합니다.
function getNextFolderColor(existingFolders: Folder[]): string { const allColors = Object.keys(FOLDER_COLOR_MAP); const usedColors = existingFolders.map(f => f.color); // 1. Try unused color first const unused = allColors.filter(c => !usedColors.includes(c)); if (unused.length > 0) return unused[0]; // 2. Fall back to least-used color const counts = allColors.map(c => ({ color: c, count: usedColors.filter(u => u === c).length })); counts.sort((a, b) => a.count - b.count); return counts[0].color; }
